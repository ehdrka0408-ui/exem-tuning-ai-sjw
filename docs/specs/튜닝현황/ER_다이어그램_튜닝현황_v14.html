group_id / llm_id / plan_hash 기반 JOIN 예시
-- 1. 요청 그룹 + 소속 요청 + LLM 모델 + 요청자 (llm/user 는 group 기준) SELECT g.request_group_name, g.request_source, g.request_count, r.request_id, r.status, r.alias, m.provider, m.model_name, u.email, u.name FROM tuning_request_group g JOIN tuning_requests r ON r.group_id = g.group_id LEFT JOIN llm_models m ON m.llm_id = g.llm_id LEFT JOIN users u ON u.user_id = g.user_id WHERE g.instance_id = 1 ORDER BY g.created_at DESC, r.request_id; -- 2. 성능 ↔ 실행계획 논리 조인 (instance_id + plan_hash) SELECT p.request_id, p.phase, p.elapsed_time_sec, p.buffer_gets_count, pl.plan_hash, pl.cost FROM sql_performance p LEFT JOIN sql_plans pl ON pl.instance_id = p.instance_id AND pl.plan_hash = p.plan_hash AND pl.phase = p.phase WHERE p.request_id = 167; -- 3. Before vs After 성능 비교 (개선율) — latency 단위 Sec SELECT MAX(CASE WHEN phase='before' THEN elapsed_time_sec END) AS before_sec, MAX(CASE WHEN phase='after' THEN elapsed_time_sec END) AS after_sec, ROUND( (MAX(CASE WHEN phase='before' THEN elapsed_time_sec END) - MAX(CASE WHEN phase='after' THEN elapsed_time_sec END)) / NULLIF(MAX(CASE WHEN phase='before' THEN elapsed_time_sec END),0) * 100 , 2) AS improvement_pct FROM sql_performance WHERE request_id = 66 AND instance_id = 1; -- 4. 사용자 + 권한그룹 + 튜닝 통계 (user_id 는 group 기준 집계) SELECT u.email, u.name, ag.group_name, COUNT(r.request_id) AS total_cnt, SUM(CASE WHEN r.status='completed' THEN 1 ELSE 0 END) AS success_cnt, ROUND(AVG(r.latency_ms)::numeric / 1000, 3) AS avg_latency_sec FROM users u JOIN groups ag ON ag.group_id = u.group_id LEFT JOIN tuning_request_group g ON g.user_id = u.user_id LEFT JOIN tuning_requests r ON r.group_id = g.group_id GROUP BY u.user_id, u.email, u.name, ag.group_name ORDER BY total_cnt DESC; -- 5. LLM 모델별 비용·호출 집계 (llm_models 기준) SELECT m.provider, m.model_name, COUNT(l.request_id) AS call_cnt, SUM(l.input_tokens + l.output_tokens) AS total_tokens, ROUND(AVG(l.latency_ms)::numeric / 1000, 3) AS avg_latency_sec FROM llm_models m LEFT JOIN llm_call_log l ON l.provider = m.provider AND l.model = m.model_name GROUP BY m.llm_id, m.provider, m.model_name ORDER BY call_cnt DESC; -- 6. 요청 + AS-IS SQL + schema_name (schema_name 은 tuning_requests 로 이동) SELECT r.request_id, r.schema_name, r.result_match, s.sql_text AS asis_sql FROM tuning_requests r JOIN sql_texts s ON s.sql_id = r.asis_sql_id AND s.instance_id = r.instance_id AND s.sql_type = 'as_is' WHERE r.instance_id = 1 ORDER BY r.request_id DESC; -- 7. 정책 조회 (JSONB value 추출) SELECT key, value, description FROM policy_settings WHERE key LIKE 'global.%' ORDER BY key; -- 8. 바인드 변수 포함 AS-IS SQL 전체 컨텍스트 SELECT r.request_id, s.sql_text, b.name AS bind_name, b.data_type, b.value, b.position FROM tuning_requests r JOIN sql_texts s ON s.sql_id = r.asis_sql_id AND s.instance_id = r.instance_id AND s.sql_type = 'as_is' LEFT JOIN sql_bind_variables b ON b.request_id = r.request_id AND b.instance_id = r.instance_id AND b.sql_id = r.asis_sql_id WHERE r.request_id = 167 ORDER BY b.position; -- 9. (v14) legacy_console_id 기반 구 console_users 매핑 확인 SELECT user_id, email, name, legacy_console_id FROM users WHERE legacy_console_id IS NOT NULL ORDER BY user_id;
