-- 1. 튜닝 요청 + AS-IS SQL + Before 성능 (sql_id 매칭 추가) SELECT r.request_id, r.status, s.sql_text AS asis_sql, p.elapsed_time_sec, p.buffer_gets_count FROM tuning_requests r JOIN sql_texts s ON s.sql_id = r.asis_sql_id AND s.instance_id = r.instance_id AND s.sql_type = 'as_is' JOIN sql_performance p ON p.request_id = r.request_id AND p.instance_id = r.instance_id AND p.sql_id = r.asis_sql_id AND p.phase = 'before' WHERE r.instance_id = 1 AND r.status = 'completed'; -- 2. Before vs After 성능 비교 (개선율 계산) — 실측 92.6% 개선 SELECT MAX(CASE WHEN phase='before' THEN elapsed_time_sec END) AS before_sec, MAX(CASE WHEN phase='after' THEN elapsed_time_sec END) AS after_sec, ROUND( (MAX(CASE WHEN phase='before' THEN elapsed_time_sec END) - MAX(CASE WHEN phase='after' THEN elapsed_time_sec END)) / NULLIF(MAX(CASE WHEN phase='before' THEN elapsed_time_sec END),0) * 100 , 2) AS improvement_pct FROM sql_performance WHERE request_id = 66 AND instance_id = 1; -- 3. 재시도 이력 (같은 AS-IS SQL의 모든 시도) SELECT request_id, parent_request_id, status, rationale, requested_at FROM tuning_requests WHERE asis_sql_id = '1e6c05b8c1bd33a0f5f007e85b30a940' AND instance_id = 1 ORDER BY request_id DESC; -- 4. 인스턴스별 튜닝 현황 집계 (db_instances PK 컬럼명 instance_id 사용) SELECT r.instance_id, i.name, i.alias, COUNT(*) AS total_cnt, SUM(CASE WHEN r.status='completed' THEN 1 ELSE 0 END) AS success_cnt, ROUND(AVG(r.latency_ms)::numeric / 1000, 3) AS avg_latency_sec FROM tuning_requests r JOIN db_instances i ON i.instance_id = r.instance_id GROUP BY r.instance_id, i.name, i.alias; -- 5. LLM 비용/성능 분석 (요청 + 호출로그 결합, latency 단위 Sec 환산) SELECT r.request_id, r.llm_provider, r.llm_model, r.input_tokens, r.output_tokens, ROUND(r.latency_ms::numeric / 1000, 3) AS latency_sec, l.cached_tokens, l.cache_creation_tokens, l.status FROM tuning_requests r LEFT JOIN llm_call_log l ON l.request_id = r.request_id WHERE r.instance_id = 1 ORDER BY r.requested_at DESC; -- 6. 바인드 변수 포함 AS-IS SQL 전체 컨텍스트 (sql_id 매칭 추가) SELECT r.request_id, s.sql_text, b.name AS bind_name, b.data_type, b.value, b.position FROM tuning_requests r JOIN sql_texts s ON s.sql_id = r.asis_sql_id AND s.instance_id = r.instance_id AND s.sql_type = 'as_is' LEFT JOIN sql_bind_variables b ON b.request_id = r.request_id AND b.instance_id = r.instance_id AND b.sql_id = r.asis_sql_id WHERE r.request_id = 66 ORDER BY b.position; -- 7. 추정값 제외한 실측 성능만 집계 (is_estimated 활용) SELECT request_id, phase, elapsed_time_sec, cpu_time_sec, buffer_gets_count, result_match FROM sql_performance WHERE instance_id = 1 AND is_estimated = 'N' ORDER BY captured_at DESC; -- 8. 사용자별 튜닝 요청 통계 (user_id 기반, User 도메인 확장 대비) SELECT r.user_id, COUNT(*) AS total_cnt, SUM(CASE WHEN r.status='completed' THEN 1 ELSE 0 END) AS success_cnt, ROUND(AVG(r.latency_ms)::numeric / 1000, 3) AS avg_latency_sec FROM tuning_requests r WHERE r.instance_id = 1 AND r.user_id IS NOT NULL GROUP BY r.user_id ORDER BY total_cnt DESC;
